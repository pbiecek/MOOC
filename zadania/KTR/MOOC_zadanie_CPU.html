Clock <- as.character(dane$Clock)
Clock <- as.numeric(paste(substr(Clock,1,4)))
Cores <- as.character(dane$Cores)
Cores <- as.numeric(gsub(pattern = "-Core", replacement = "",  x = Cores))
Process <- as.character(dane$Process)
Process <- strsplit(Process, " ")
Process <- as.numeric(sapply(1:length(Process),function(x) Process[[x]][1]))
TDP <- as.character(dane$TDP)
TDP <- as.numeric(gsub(pattern = "W", replacement = "",  x = TDP))
tab <- data.frame(Cores,Process,TDP,Clock=Clock,Classification=dane$Classification,Manufacturer=dane$Manufacturer,p=dane$Process)
