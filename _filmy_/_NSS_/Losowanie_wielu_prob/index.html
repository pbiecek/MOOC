## Symulowanie zwiêkszania prób
library(RCurl)
library(bitops)
library(XML)
library(knitr)
library(ggplot2)
library(animation)
ani.options(interval = 0.05, nmax = 18)
par(mar = c(3, 3, 2, 0.5), mgp = c(2, 0.5, 0), tcl = -0.3, 
    cex.axis = 0.8, cex.lab = 0.8, cex.main = 1)
for (i in 1:ani.options("nmax")) {
    hist(proba[[i]], xlim = c(0, 1), ylim = c(0, 500), xlab = paste("Wielkoœæ próby:", 
        c(i + 1), sep = " "), col = "black")
    abline(v = 7/20, col = "red")
    ani.pause(1)
}
